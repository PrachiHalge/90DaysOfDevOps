#!/bin/bash

echo "#90DaysOfDevopsChallenge"
echo "Enter One Number"

read  A
echo "First Number is:$A"

echo "You have given second number as argument is $1 "

if  [ $A -gt $1 ]
then
        echo "$A is greater than $1"
else
        echo "$1 is greater that $A"
fi
